import { Progress, PROGRESS_DEFAULTS } from "../progress";

# Progress Bar

Native HTML **progress** bars.

<Playground>
  <Progress value={15} max={100} />
</Playground>

### Colors

Use the `color` prop on `<Progress>` to set the color.

<Playground>
  {() =>
    DEFAULTS.colors.map((color, i) => (
      <OptionBlock props={{ color }} index={i} key={i}>
        <Progress color={color} max={100} value={20 + i * 5} />
      </OptionBlock>
    ))
  }
</Playground>

### Sizes

Use the `size` prop of `<Progress>` to change the height.

<Playground>
  {() => {
    const sizes = PROGRESS_DEFAULTS.sizes.map(size => ({ name: size, size }));
    sizes.splice(1, 0, { name: "-", size: undefined });

    return sizes.map(({ name, size }, i) => (
      <OptionBlock props={{ size }} index={i} key={i}>
        <Progress value={20 + i * 10} max={100} size={size} />
      </OptionBlock>
    ))

}}

</Playground>

<ForwardRefAsExoticComponentDoc
